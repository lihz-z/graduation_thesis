\begin{notation}

  \begin{notationlist}{6em}
    \item[$\displaystyle \Omega_r$] 雨滴遮挡区域
    \item[$\displaystyle I_{\text{obs}}$] 雨天条件下的观测图像灰度值
    \item[$\displaystyle I_{\text{scene}}$] 无遮挡条件下的场景真实灰度值
    \item[$\displaystyle \theta_i, \theta_t$] 水膜反射模型中的入射角与折射角
    \item[$\displaystyle I_{\text{syn}}$] 雨纹合成后的增强图像
    \item[$\displaystyle \gamma$] 数据增强强度等级
    \item[$\displaystyle H_r$] CLRNet 中的行锚点数量
    \item[$\displaystyle \hat{x}_{ij}, x_{ij}$] 第 $i$ 条车道线在第 $j$ 行的预测与真实水平位置
    \item[$\displaystyle \hat{y}_{ij}, y_{ij}$] 第 $i$ 条车道线在第 $j$ 行的存在性预测概率与真实标签
    \item[$\displaystyle \lambda$] 雨纹混合或特征残差融合中的权重系数
    \item[$\displaystyle \mathcal{L}_{\text{pos}}, \mathcal{L}_{\text{cls}}, \mathcal{L}_{\text{total}}$] 位置损失、分类损失与总损失函数
  \end{notationlist}

\end{notation}
